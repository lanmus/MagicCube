const db = require('../models/db');
 
class ProductController {
    // 获取商品详情
    async getDetail(req, res) {
        try {
            const productId = req.params.id;
 
            // 获取商品基本信息
            const [products] = await db.query(
                'SELECT * FROM products WHERE id = ?',
                [productId]
            );
 
            if (products.length === 0) {
                return res.status(404).json({ error: '商品不存在' });
            }
 
            const product = products[0];
 
            // 获取所有模块
            const [modules] = await db.query(
                'SELECT * FROM modules WHERE product_id = ? ORDER BY module_type, sort_order',
                [productId]
            );
 
            // 获取所有图片
            const [images] = await db.query(
                'SELECT i.* FROM images i ' +
                'JOIN modules m ON i.module_id = m.id ' +
                'WHERE m.product_id = ? ' +
                'ORDER BY m.sort_order, i.sort_order',
                [productId]
            );
 
            // 组织数据结构
            const moduleMap = new Map();
            modules.forEach(module => {
                module.images = images.filter(img => img.module_id === module.id);
                moduleMap.set(module.id, module);
            });
 
            product.modules = Array.from(moduleMap.values());
 
            res.json(product);
        } catch (error) {
            res.status(500).json({ error: error.message });
        }
    }
 
    // 获取商品列表
    async list(req, res) {
        try {
            const { 
                page = 1, 
                pageSize = 10,
                gender,
                ageRange,
                scene,
                style,
                search
            } = req.query;
 
            const offset = (page - 1) * pageSize;
            let query = 'SELECT * FROM products WHERE 1=1';
            const params = [];
 
            // 添加筛选条件
            if (gender) {
                query += ' AND gender = ?';
                params.push(gender);
            }
            if (ageRange) {
                query += ' AND age_range = ?';
                params.push(ageRange);
            }
            if (scene) {
                query += ' AND scene = ?';
                params.push(scene);
            }
            if (style) {
                query += ' AND style = ?';
                params.push(style);
            }
            if (search) {
                query += ' AND (name LIKE ? OR spu_code LIKE ?)';
                params.push(`%${search}%`, `%${search}%`);
            }
 
            // 获取总数
            const [countResult] = await db.query(
                `SELECT COUNT(*) as total FROM (${query}) as t`,
                params
            );
            const total = countResult[0].total;
 
            // 添加分页
            query += ' ORDER BY created_at DESC LIMIT ? OFFSET ?';
            params.push(parseInt(pageSize), offset);
 
            const [products] = await db.query(query, params);
 
            // 获取每个商品的主图
            const productIds = products.map(p => p.id);
            if (productIds.length > 0) {
                const [mainImages] = await db.query(
                    `SELECT i.*, m.product_id 
                     FROM images i 
                     JOIN modules m ON i.module_id = m.id 
                     WHERE m.product_id IN (?) 
                     AND m.module_type = 'main' 
                     AND m.sub_type = 'main_image'
                     ORDER BY i.sort_order 
                     LIMIT 1`,
                    [productIds]
                );
 
                // 将主图添加到对应的商品中
                products.forEach(product => {
                    const mainImage = mainImages.find(img => img.product_id === product.id);
                    product.mainImage = mainImage || null;
                });
            }
 
            res.json({
                total,
                page: parseInt(page),
                pageSize: parseInt(pageSize),
                products
            });
        } catch (error) {
            res.status(500).json({ error: error.message });
        }
    }
 
    // 搜索商品
    async search(req, res) {
        try {
            const { keyword, page = 1, pageSize = 10 } = req.query;
            const offset = (page - 1) * pageSize;
 
            let query = `
                SELECT p.*, 
                       COUNT(*) OVER() as total_count
                FROM products p
                WHERE p.name LIKE ? 
                   OR p.spu_code LIKE ?
                   OR p.scene LIKE ?
                   OR p.style LIKE ?
            `;
            const searchPattern = `%${keyword}%`;
            const params = [searchPattern, searchPattern, searchPattern, searchPattern];
 
            // 添加分页
            query += ' ORDER BY p.created_at DESC LIMIT ? OFFSET ?';
            params.push(parseInt(pageSize), offset);
 
            const [results] = await db.query(query, params);
            
            const total = results.length > 0 ? results[0].total_count : 0;
            const products = results.map(({ total_count, ...product }) => product);
 
            res.json({
                total,
                page: parseInt(page),
                pageSize: parseInt(pageSize),
                products
            });
        } catch (error) {
            res.status(500).json({ error: error.message });
        }
    }
 
    // 创建商品
    async create(req, res) {
        try {
            const { name, spu_code, gender, age_range, scene, style, designer_3d, designer_2d } = req.body;
            
            // 创建商品
            const product = await db.query(
                `INSERT INTO products (name, spu_code, gender, age_range, scene, style, designer_3d, designer_2d, user_id)
                 VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
                 RETURNING *`,
                [name, spu_code, gender, age_range, scene, style, designer_3d, designer_2d, req.user.id]
            );
 
            // 创建默认模块
            const module = await db.query(
                `INSERT INTO modules (product_id, name, module_type, sub_type)
                 VALUES ($1, $2, $3, $4)
                 RETURNING *`,
                [product.rows[0].id, '默认模块', 'main', 'default']
            );
 
            // 返回包含模块的商品数据
            res.status(201).json({
                ...product.rows[0],
                modules: [module.rows[0]]
            });
        } catch (error) {
            console.error('创建商品失败:', error);
            res.status(500).json({ error: '创建商品失败' });
        }
    }
 
    // 更新商品
    async update(req, res) {
        const connection = await db.beginTransaction();
        try {
            const productId = req.params.id;
            const {
                name,
                spuCode,
                gender,
                ageRange,
                scene,
                style,
                designer3d,
                designer2d,
                modules
            } = req.body;
 
            // 1. 检查商品是否存在
            const [products] = await connection.query(
                'SELECT * FROM products WHERE id = ?',
                [productId]
            );
 
            if (products.length === 0) {
                await connection.rollback();
                return res.status(404).json({ error: '商品不存在' });
            }
 
            // 2. 更新商品基本信息
            await connection.query(
                `UPDATE products SET 
                name = ?, spu_code = ?, gender = ?, age_range = ?, 
                scene = ?, style = ?, designer_3d = ?, designer_2d = ?
                WHERE id = ?`,
                [name, spuCode, gender, ageRange, scene, style, designer3d, designer2d, productId]
            );
 
            // 3. 更新模块
            if (modules && modules.length > 0) {
                // 删除旧的模块
                await connection.query('DELETE FROM modules WHERE product_id = ?', [productId]);
 
                // 创建新的模块
                for (const module of modules) {
                    const [moduleResult] = await connection.query(
                        `INSERT INTO modules 
                        (product_id, module_type, sub_type, name, sort_order) 
                        VALUES (?, ?, ?, ?, ?)`,
                        [productId, module.type, module.subType, module.name, module.sortOrder]
                    );
                    module.id = moduleResult.insertId;
                }
            }
 
            await connection.commit();
            res.json({
                success: true,
                modules: modules
            });
        } catch (error) {
            await connection.rollback();
            res.status(500).json({ error: error.message });
        }
    }
 
    // 删除商品
    async delete(req, res) {
        const connection = await db.beginTransaction();
        try {
            const productId = req.params.id;
            const userId = req.user.id;
 
            // 检查商品是否存在且属于当前用户
            const [product] = await connection.query(
                'SELECT * FROM products WHERE id = ? AND user_id = ?',
                [productId, userId]
            );
 
            if (!product || product.length === 0) {
                await connection.rollback();
                return res.status(404).json({ error: '商品不存在或无权限删除' });
            }
 
            // 删除相关的图片记录
            await connection.query(`
                DELETE images FROM images 
                INNER JOIN modules ON images.module_id = modules.id 
                WHERE modules.product_id = ?
            `, [productId]);
 
            // 删除相关的模块
            await connection.query(
                'DELETE FROM modules WHERE product_id = ?',
                [productId]
            );
 
            // 删除商品
            await connection.query(
                'DELETE FROM products WHERE id = ?',
                [productId]
            );
 
            await connection.commit();
            res.json({ success: true });
        } catch (error) {
            await connection.rollback();
            console.error('删除商品失败:', error);
            res.status(500).json({ error: error.message });
        } finally {
            connection.release();
        }
    }
